SELECT LASTNAME, SALARY, DECIMAL(ROUND(((SALARY*3/100)+SALARY),2),8,2)   
AS SALARY_Y_INC,DECIMAL(ROUND((((SALARY*3/100)+SALARY)/12),2),8,2) AS          
SALARY_M_INC FROM GEN_EMPLOYEE 
WHERE ((SALARY*3/100)+SALARY) >=40000 
ORDER BY SALARY;
